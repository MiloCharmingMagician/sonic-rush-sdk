Sonic Rush AMA Format
By Milo Charming Magician

struct_root
{
    Header header;
    EntryTable etb[entryCount];
    Entry entry[entryCount];
};

struct_Header
{
    u32 magic; // #AMA
    u32 entryCount;
    u32 fixed; // always '16'
};

struct_EntryTable
{
    u32 unk[5];
    u32 offset;
};

struct_Entry
{
    u32 unk[2];
    u32 x;
    u32 y;
    u32 unk[4];
};
